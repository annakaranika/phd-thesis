% =====================================================================
% DRAFT / REVIEW TOGGLES
% our_comments: show colored reviewer/author annotations
% our_contents: show new/deleted text in color (blue=add, red=delete)

% =====================================================================
% SHARED THEOREM ENVIRONMENTS (CoMesh, RASC)
% Defined once here to avoid redefinition errors when both are active.
% =====================================================================
\newtheorem{definition}{Definition}
\newtheorem{lemma}{Lemma}
\newtheorem{theorem}{Theorem}
\newtheorem{corollary}{Corollary}[theorem]
\newtheorem{fact}{Fact}
\newtheorem{invariant}{Invariant}

% Number all floats, equations, and theorems as Chapter.N
% \thesection expands to "Chapter 1", so we override each \the<counter>
% to use \arabic{section} directly, giving clean "1.1" format.
\numberwithin{figure}{section}
\renewcommand{\thefigure}{\arabic{section}.\arabic{figure}}
\numberwithin{table}{section}
\renewcommand{\thetable}{\arabic{section}.\arabic{table}}
\numberwithin{equation}{section}
% \AtBeginDocument needed: tagging system patches amsmath at \begin{document}
% and would otherwise overwrite \theequation
\AtBeginDocument{\renewcommand{\theequation}{\arabic{section}.\arabic{equation}}}
\numberwithin{algorithm}{section}
\renewcommand{\thealgorithm}{\arabic{section}.\arabic{algorithm}}
\numberwithin{theorem}{section}
\renewcommand{\thetheorem}{\arabic{section}.\arabic{theorem}}
\numberwithin{lemma}{section}
\renewcommand{\thelemma}{\arabic{section}.\arabic{lemma}}
\numberwithin{definition}{section}
\renewcommand{\thedefinition}{\arabic{section}.\arabic{definition}}
\numberwithin{fact}{section}
\renewcommand{\thefact}{\arabic{section}.\arabic{fact}}
\numberwithin{invariant}{section}
\renewcommand{\theinvariant}{\arabic{section}.\arabic{invariant}}
% corollary is defined as [theorem]-parented; override to number by section instead
\numberwithin{corollary}{section}
\renewcommand{\thecorollary}{\arabic{section}.\arabic{corollary}}
